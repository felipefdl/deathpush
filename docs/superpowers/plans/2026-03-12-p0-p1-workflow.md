- [ ] **Step 1: Fix the Rust function**

In `open_repository()`, replace the manual `RepositoryStatus` construction with a call to `get_repository_status()` after storing the session:

```rust
#[uniffi::export]
pub fn open_repository(session_id: String, path: String) -> Result<RepositoryStatus, Error> {
  let mgr = manager();
  let repo = GitRepository::open(&PathBuf::from(&path))?;
  let repo_root = repo.root().to_path_buf();

  // Stop old watcher for this session, start new one
  {
    let mut watchers = mgr.watcher_state.lock().map_err(|e| Error::other(e.to_string()))?;
    watchers.remove(&session_id);
  }
  if let Some(sink) = get_event_sink() {
    if let Err(err) = watcher::start_watcher(&session_id, sink, &repo_root, &mgr.watcher_state) {
      tracing::warn!("failed to start watcher: {:?}", err);
    }
  }

  // Store session BEFORE calling get_repository_status
  let status = {
    let mut sessions = mgr.sessions.lock().map_err(|e| Error::other(e.to_string()))?;
    let state = sessions.entry(session_id.clone()).or_default();
    state.cli_root = Some(repo_root);
    state.repo = Some(repo);

    // Build status from the stored repo (which has proper groups)
    let repo_ref = state.repo.as_ref().ok_or(Error::NoRepository)?;
    get_repository_status(repo_ref)?
  };

  Ok(status)
}
```

				dismiss()
			}
		}

	private func dismiss() {
		withAnimation(.easeInOut(duration: 0.25)) {
			isVisible = false
		}
		DispatchQueue.main.asyncAfter(deadline: .now() + 0.3) {
			onDismiss()
		}
	}

	func onAppearAnimate() -> some View {
		self.onAppear {
			withAnimation(.easeInOut(duration: 0.25)) {
				isVisible = true
			}
		}
	}

```swift
import SwiftUI

struct StashDiffView: View {
	let stashIndex: UInt32
	@Environment(AppState.self) private var appState
	@Environment(TabState.self) private var tabState
	@State private var hunks: FileDiffWithHunks?
	@State private var errorMessage: String?
	@State private var selectedFileIndex: Int?

	private var repoService: RepositoryService? {
		tabState.repoService
	}

	var body: some View {
		VStack(spacing: 0) {
			stashHeader

			Divider()

			if let error = errorMessage {
				ContentUnavailableView(
					"Error Loading Stash",
					systemImage: "exclamationmark.triangle",
					description: Text(error)
				)
			} else if let hunks {
				if hunks.hunks.isEmpty {
					ContentUnavailableView(
						"Empty Stash",
						systemImage: "tray",
						description: Text("This stash contains no changes.")
					)
				} else {
					stashDiffContent(hunks)
				}
			} else {
				ProgressView("Loading stash...")
					.frame(maxWidth: .infinity, maxHeight: .infinity)
			}
		}
		.task(id: stashIndex) {
			loadStash()
		}
	}

	@ViewBuilder
	private var stashHeader: some View {
		GlassEffectContainer(spacing: 8) {
			HStack {
				HStack(spacing: 4) {
					Image(systemName: "tray.full")
						.foregroundStyle(.secondary)
					Text("stash@{\(stashIndex)}")
						.font(.callout.monospaced())
				}
				Spacer()
				Text("\(hunks?.hunks.count ?? 0) hunks")
					.font(.caption)
					.foregroundStyle(.secondary)
			}
			.padding(.horizontal, 12)
			.padding(.vertical, 6)
		}
	}

	@ViewBuilder
	private func stashDiffContent(_ diff: FileDiffWithHunks) -> some View {
		// Reconstruct as a unified diff for Monaco
		let original = diff.hunks.flatMap { hunk in
			hunk.lines.compactMap { line -> String? in
				line.lineType == "remove" || line.lineType == "context" ? line.content : nil
			}
		}.joined(separator: "\n")

		let modified = diff.hunks.flatMap { hunk in
			hunk.lines.compactMap { line -> String? in
				line.lineType == "add" || line.lineType == "context" ? line.content : nil
			}
		}.joined(separator: "\n")

		let diffContent = DiffContent(
			path: "stash@{\(stashIndex)}",
			original: original,
			modified: modified,
			originalLanguage: nil,
			fileType: "text",
			isBinary: false
		)

		MonacoDiffView(
			diff: diffContent,
			diffMode: .inline,
			themeJSON: appState.themeService.themeDataJSON
		)
		.background(Color(nsColor: appState.themeService.color(forKey: "editor.background") ?? .black))
	}

	private func loadStash() {
		errorMessage = nil
		do {
			hunks = try repoService?.showStash(index: stashIndex)
		} catch {
			errorMessage = error.localizedDescription
		}
	}
}
```

import SwiftUI

struct BlameView: View {
	let path: String
	@Environment(AppState.self) private var appState
	@Environment(TabState.self) private var tabState
	@State private var blame: FileBlame?
	@State private var fileContent: FileContent?
	@State private var errorMessage: String?
	@State private var hoveredCommitId: String?

	private var repoService: RepositoryService? {
		tabState.repoService
	}

	var body: some View {
		VStack(spacing: 0) {
			blameHeader

			Divider()

			if let error = errorMessage {
				ContentUnavailableView(
					"Error Loading Blame",
					systemImage: "exclamationmark.triangle",
					description: Text(error)
				)
			} else if let blame, let content = fileContent {
				blameContent(blame: blame, content: content)
			} else {
				ProgressView("Loading blame...")
					.frame(maxWidth: .infinity, maxHeight: .infinity)
			}
		}
		.task(id: path) {
			loadBlame()
		}
	}

	@ViewBuilder
	private var blameHeader: some View {
		GlassEffectContainer(spacing: 8) {
			HStack {
				HStack(spacing: 4) {
					Image(systemName: "person.text.rectangle")
						.foregroundStyle(.secondary)
					Text(path)
						.font(.callout)
						.lineLimit(1)
						.truncationMode(.middle)
				}
				Spacer()
				Button(action: {
					tabState.explorerShowBlame = false
				}) {
					Text("Close Blame")
				}
				.buttonStyle(.glass)
				.controlSize(.small)
			}
			.padding(.horizontal, 12)
			.padding(.vertical, 6)
		}
	}

	@ViewBuilder
	private func blameContent(blame: FileBlame, content: FileContent) -> some View {
		let lines = content.content.components(separatedBy: "\n")

		ScrollView(.vertical) {
			LazyVStack(alignment: .leading, spacing: 0) {
				ForEach(blame.lineGroups, id: \.startLine) { group in
					ForEach(group.startLine...group.endLine, id: \.self) { lineNum in
						HStack(spacing: 0) {
							// Blame gutter
							blameGutter(group: group, lineNumber: lineNum, isFirstLine: lineNum == group.startLine)

							// Separator
							Rectangle()
								.fill(Color.secondary.opacity(0.2))
								.frame(width: 1)

							// Line number
							Text("\(lineNum)")
								.font(.system(size: 11, design: .monospaced))
								.foregroundStyle(.tertiary)
								.frame(width: 40, alignment: .trailing)
								.padding(.trailing, 8)

							// Code
							let lineIndex = Int(lineNum) - 1
							Text(lineIndex < lines.count ? lines[lineIndex] : "")
								.font(.system(size: 13, design: .monospaced))
								.lineLimit(1)
								.frame(maxWidth: .infinity, alignment: .leading)
						}
						.background(hoveredCommitId == group.commitId ? Color.accentColor.opacity(0.05) : Color.clear)
						.onHover { hovering in
							hoveredCommitId = hovering ? group.commitId : nil
						}
						.onTapGesture {
							// Navigate to commit in history
							tabState.sidebarSelection = .history
							tabState.selectedCommitId = group.commitId
						}
					}
				}
			}
		}
		.background(Color(nsColor: appState.themeService.color(forKey: "editor.background") ?? .black))
	}

	@ViewBuilder
	private func blameGutter(group: BlameLineGroup, lineNumber: UInt32, isFirstLine: Bool) -> some View {
		VStack(alignment: .leading, spacing: 0) {
			if isFirstLine {
				Text(group.authorName)
					.font(.system(size: 11).bold())
					.lineLimit(1)
				HStack(spacing: 4) {
					Text(group.shortId)
						.font(.system(size: 10, design: .monospaced))
					Text(formatRelativeDate(group.authorDate))
						.font(.system(size: 10))
				}
				.foregroundStyle(.tertiary)
			}
		}
		.frame(width: 180, alignment: .leading)
		.padding(.horizontal, 8)
		.padding(.vertical, 1)
	}

	private func formatRelativeDate(_ dateString: String) -> String {
		// The date comes as ISO 8601 from Rust
		let formatter = ISO8601DateFormatter()
		formatter.formatOptions = [.withInternetDateTime]
		guard let date = formatter.date(from: dateString) else { return dateString }
		let relative = RelativeDateTimeFormatter()
		relative.unitsStyle = .abbreviated
		return relative.localizedString(for: date, relativeTo: Date())
	}

	private func loadBlame() {
		errorMessage = nil
		do {
			blame = try repoService?.blameFile(path: path)
			fileContent = try repoService?.readExplorerFile(path: path)
		} catch {
			errorMessage = error.localizedDescription
		}
	}
}

- [ ] **Step 1: Add WindowCloseGuard NSViewRepresentable**

In `ContentView.swift`, add a new struct after `WindowTabbingConfigurator`:

```swift
private struct WindowCloseGuard: NSViewRepresentable {
	let hasUnsavedChanges: () -> Bool

	func makeNSView(context: Context) -> NSView {
		let view = NSView()
		context.coordinator.view = view
		DispatchQueue.main.async {
			guard let window = view.window else { return }
			context.coordinator.originalDelegate = window.delegate
			window.delegate = context.coordinator
		}
		return view
	}

	func updateNSView(_ nsView: NSView, context: Context) {
		context.coordinator.hasUnsavedChanges = hasUnsavedChanges
	}

	func makeCoordinator() -> Coordinator {
		Coordinator()
	}

	class Coordinator: NSObject, NSWindowDelegate {
		weak var view: NSView?
		weak var originalDelegate: (any NSWindowDelegate)?
		var hasUnsavedChanges: (() -> Bool)?

		func windowShouldClose(_ sender: NSWindow) -> Bool {
			guard hasUnsavedChanges?() == true else { return true }

			let alert = NSAlert()
			alert.messageText = "You have uncommitted changes"
			alert.informativeText = "Closing this window will not discard your changes, but you may lose track of them."
			alert.alertStyle = .warning
			alert.addButton(withTitle: "Close Anyway")
			alert.addButton(withTitle: "Cancel")

			let response = alert.runModal()
			return response == .alertFirstButtonReturn
		}

		// Forward other delegate methods to original
		func windowWillClose(_ notification: Notification) {
			originalDelegate?.windowWillClose?(notification)
		}

		func windowDidBecomeKey(_ notification: Notification) {
			originalDelegate?.windowDidBecomeKey?(notification)
		}

		func windowDidResignKey(_ notification: Notification) {
			originalDelegate?.windowDidResignKey?(notification)
		}
	}
}
```

The `stageSelectedLines` function needs to map Monaco line numbers back to hunk indices and line offsets. This requires loading the hunks data first:

```swift
@State private var hunksData: FileDiffWithHunks?

private func loadDiff(for filePath: String) {
	// ... existing code ...
	// Also load hunks for line staging
	hunksData = try? repoService?.getHunks(path: filePath, staged: isStaged)
}

private func stageSelectedLines(start: Int, end: Int) {
	guard let hunks = hunksData else { return }
	// Find which hunk contains these lines and map to hunk-relative indices
	// This requires mapping Monaco editor line numbers to hunk line indices
	// The hunk's DiffLine array indices are what stageLinesInHunk expects
	for (hunkIdx, hunk) in hunks.hunks.enumerated() {
		let hunkStart = Int(hunk.newStart)
		let hunkEnd = hunkStart + Int(hunk.newLines) - 1
		if start <= hunkEnd && end >= hunkStart {
			let lineStart = max(0, start - hunkStart)
			let lineEnd = min(Int(hunk.lines.count) - 1, end - hunkStart)
			try? repoService?.stageLinesInHunk(
				path: path,
				hunkIndex: UInt32(hunkIdx),
				lineStart: UInt32(lineStart),
				lineEnd: UInt32(lineEnd),
				staged: isStaged
			)
			diffVersion += 1
			loadDiff(for: path)
			selectedLineStart = nil
			selectedLineEnd = nil
			return
		}
	}
}
```

Note: The line number mapping between Monaco's diff editor lines and the hunk's `DiffLine` array indices is non-trivial. Monaco's modified editor shows the "new" side, so line numbers correspond to `newStart` + offset within the hunk. However, context lines shift things. You may need to pass hunk metadata to JS and have JS report which hunk a selection falls in, rather than mapping on the Swift side. Consider sending `hunkRanges` to `enableLineSelection()` and having JS report `{hunkIndex, lineStart, lineEnd}` directly.

## Important Notes

### Build & Test
- After Rust changes: `cargo check -p deathpush-ffi && cargo test -p deathpush-core`
- After Swift changes: Build via Xcode (`Cmd+B`) -- all UniFFI types are only available after the Rust build step
- SourceKit errors about missing types (e.g., `RepositoryStatus`, `DiffContent`) are expected in the editor; they resolve when Xcode builds the full pipeline

### UniFFI Generated Types
- Never edit `DeathPush/Bridge/Generated/DeathPushCore.swift` or `deathpush_core.swift`
- If you need to know a type's initializer signature, read the generated file
- `DiffContent` is a Rust struct exposed via UniFFI -- check if it has a public memberwise init in the generated Swift code before constructing it manually

### Liquid Glass UI
- Use `.buttonStyle(.glass)` for secondary buttons, `.glassProminent` for primary
- Use `GlassEffectContainer` for grouped controls (headers, toolbars)
- Use `.glassEffect(.regular)` for badge/pill elements

### Parallelization
Tasks 1-3 (P0) can all be done in parallel -- they touch independent files.
Tasks 4-7 (P1) have some overlap in `TabState.swift` and `ContentView.swift`:
- Tasks 4 and 5 both modify `TabState.swift` -- merge carefully
- Tasks 3 and 6 both modify `ContentView.swift` -- merge carefully
- Task 7 is fully independent (Monaco + DiffDetailView)

Recommended execution order for minimal conflicts:
1. Task 1 (Rust, isolated)
2. Tasks 2 + 3 in parallel (both Swift, different files except ContentView -- Task 3 modifies ContentView)
3. Tasks 4 + 5 together (both need TabState changes)
4. Task 6 (ContentView, after Task 3 is merged)
5. Task 7 (Monaco + DiffDetailView, isolated)
